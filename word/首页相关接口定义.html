首页相关接口定义

1.首页精品列表

接口名称：getHomeImage

取图片

postData:

    {
    }

getData:

    {   
        
        /*
        * 这个里取的是后台广告里的图片，最多5张图片，可左右滑动查看
        */
        ImageList: [
            {
                ImageUrl: //-----图片地址
                TargetModel: //--跳转类型（这个参数待定）
                H5Url: //--------H5点击跳转到的网址
                AppUrl: //-------APP点击跳转到的网址
            }
        ]
    }

接口名称：getHomeList

取最新推荐的帖子可分页

postData:

    {
        PageIndex: //--当前页码
        PageSize: //---每页显示多少条记录
    }

getData:

    {   
        
        /*
        * 帖子展示以瀑布流的形式，向下滑动可无限加载，首页呈现的均为编辑在后台推荐的帖子。
        */
        ArticleList: [
            {
                ImageUrl: //----------图片地址
                Content: //-----------描述内容
                CollectNum: //--------被收藏的数量
                UserId: //------------用户ID
                UserImage: //---------用户头像 
                UserName: //----------用户名称
                Category:[ //---------帖子所属栏目名称（一级、二级、三级名称）不会跨栏目
                    {
                        CateId: //----栏目ID
                        CateName: //--栏目名称
                    }
                ]
            }
        ]
    }

2.首页关注列表

接口名称：getHomeFollow

关注调取自己发布的及关注过的好友发布的帖子，帖子展示以瀑布流的形式，向下滑动可无限加载。

postData:

    {   
        PageIndex: //---当前页码
        PageSize: //----每页显示多少条记录

        Header: {
            UserId: //--当前登录用户的ID
            Cid: //-----设备Id
            Auth: //----当前登录用户的Auth
        }
    }

getData:

    {   
        ArticleList: [
            {
                ImageUrl: //----------图片地址
                Content: //-----------描述内容
                CollectNum: //--------被收藏的数量
                UserId: //------------用户ID
                UserImage: //---------用户头像 
                UserName: //----------用户名称
                Category:[ //---------帖子所属栏目名称（一级、二级、三级名称）不会跨栏目
                    {
                        CateId: //----栏目ID
                        CateName: //--栏目名称
                    }
                ]
            }
        ]
    }
