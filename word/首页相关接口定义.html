首页相关接口定义

1.首页精品列表

接口名称：getHomeImage

取图片

postData:

    {

        Header: {
            UserId: //--当前登录用户ID（未登录传空）
            Auth: //----当前登录用户Auth（未登录传空）
        }
    }

getData:

    {   
        
        /*
        * 这个里取的是后台广告里的图片，最多5张图片，可左右滑动查看
        */
        ImageList: [
            {
                Title: //--------标题
                ImageUrl: //-----图片地址
                TargetModel: //--跳转类型（这个参数待定）
                H5Url: //--------H5点击跳转到的网址
                AppUrl: //-------APP点击跳转到的网址
            }
        ]


        此处格式固定，服务器返回验证数据
        Response: {
            Time: //--------------服务器当前时间
            State: //-------------用户登录状态（200：用户登录成功；-200：用户登录失败或者未登录）
        }
    }

2.首页关注列表

接口名称：getHomeFollow

关注调取自己发布的及关注过的好友发布的帖子，帖子展示以瀑布流的形式，向下滑动可无限加载。

postData:

    {   
        UserId: //------用户ID

        PageIndex: //---当前页码
        PageSize: //----每页显示多少条记录

        Header: {
            UserId: //--当前登录用户ID（未登录传空）
            Auth: //----当前登录用户Auth（未登录传空）
        }
    }

getData:

    {   
        ArticleList: [
            {
                ArticleId: //------------帖子ID
                Images: [ //-------------帖子里的图片
                    {
                        ImageUrl: //-----帖子图片
                        Description: //--帖子描述
                    }
                ]

                Author: {
                    UserId: //-----------帖子发布者ID
                    ImageUrl: //---------帖子发布者头像
                    UserName: //---------帖子发布者名称
                }

                
                /*
                * 帖子所属栏目名称（一级、二级、三级名称）不会跨栏目，这个栏目是用户选择的
                */
                CategoryList: [
                    {
                        CateId: //-------栏目ID
                        CateName: //-----栏目名称
                    }
                ]

                TotalCollect: //---------帖子被收藏的总数
            }
        ]
        Total: //------------------------帖子总数


        此处格式固定，服务器返回验证数据
        Response: {
            Time: //---------------------服务器当前时间
            State: //--------------------用户登录状态（200：用户登录成功；-200：用户登录失败或者未登录）
        }
    }

3.帖子列表 （2015-06-12 新增）

接口名称：getHomeArticle

首页的帖子是通过 推荐 的帖子才能显示 （这里的帖子是来自所有栏目）

postData:

    {
        PageIndex: //---当前页码
        PageSize: //----每页显示多少条记录

        Header: {
            UserId: //--当前登录用户ID（未登录传空）
            Auth: //----当前登录用户Auth（未登录传空）
        }
    }

getData:

    {   
        ArticleList: [
            {
                ArticleId: //------------帖子ID
                Images: [ //-------------帖子里的图片
                    {
                        ImageUrl: //-----帖子图片
                        Description: //--帖子描述
                        Width: //--------切割出来的图片的宽度
                        Height: //-------切割出来的图片的高度
                    }
                ]

                Author: {
                    UserId: //-----------帖子发布者ID
                    ImageUrl: //---------帖子发布者头像
                    UserName: //---------帖子发布者名称
                }

                
                /*
                * 帖子所属栏目名称（一级、二级、三级名称）不会跨栏目，这个栏目是用户选择的
                */
                CategoryList: [
                    {
                        CateId: //-------栏目ID
                        CateName: //-----栏目名称
                    }
                ]

                TotalCollect: //---------帖子被收藏的总数
            }
        ]
        Total: //------------------------帖子总数


        此处格式固定，服务器返回验证数据
        Response: {
            Time: //---------服务器当前时间
            State: //--------用户登录状态（200：用户登录成功；-200：用户登录失败或者未登录）
        }
    }
