栏目相关接口定义

1.栏目列表

接口名称：getListCategory

postData:

    {
        CateId: //----栏目ID（当CateId为零或者空时表示调用一级栏目）

        Header: {
            UserId: //--当前登录用户ID（未登录传空）
            Auth: //----当前登录用户Auth（未登录传空）
        }
    }

getData:

    {   
        /*
        * 当前栏目的信息
        */
        CurrentCate: {
            CateId: //-------栏目ID
            CateName: //-----栏目名称
            ImageUrl: //-----栏目封面图
            Description: //--栏目描述
            RelatedClub: //--相关圈子
        }


        /*
        * 如果CateId为空或者0时，CategoryList显示的将是一级栏目列表，如果不为空，就显示该栏目下的子栏目列表
        */
        CategoryList: [
            {
                CateId: //-------栏目ID
                CateName: //-----栏目名称
                ImageUrl: //-----栏目封面图
                Description: //--栏目描述
                RelatedClub: //--相关圈子
            }
        ]
    }
