栏目相关接口定义

1.栏目列表

接口名称：getListCategory

postData:

    {
        CateId: //----栏目ID（调用一级栏目不需要CateId）
        CateType: //--1、显示当前栏目下的子栏目(默认类型为1); 2、显示该栏目下的子栏目(包括子栏目下的子栏目）
    }

getData:

    {   
        CategoryList: [
            {
                CateId: //-------栏目ID
                CateName: //-----栏目名称
                ImageUrl: //-----栏目封面图
                Description: //--栏目描述
                RelatedClub: //--相关圈子
            }
        ]
    }
