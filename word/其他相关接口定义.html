其他相关接口定义

1.聊天接口定义

接口名称：getListChat

postData:

    {

        Header: {
            UserId: //--当前登录用户ID（未登录传空）
            Auth: //----当前登录用户Auth（未登录传空）
        }
    }

getData:

    {   
        ChatList: [
            {
                UpdateTime: //--聊天发表时间
                Content: //-----聊天内容
            }
        ]
    }
