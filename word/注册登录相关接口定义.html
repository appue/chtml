注册登录模块相关接口定义

1.注册-创建资料

接口名称：setRegInfo

postData:

    {
        UserName: //--用户昵称(用户名支持汉字、字母、数字、-和)
        Sex: //-------用户性别（男、女）
        Aera: //------地区
        Job: //-------岗位
    }

getData:

    {



        此处格式固定，服务器返回验证数据
        Response: {
            Time: //---------服务器当前时间
            State: //--------用户登录状态（200：用户登录成功；-200：用户登录失败或者未登录）
        }
    }
