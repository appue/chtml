注册登录模块相关接口定义

1.登录

接口名称：getLogin（手机号登录）

postData:

    {
        Phone: //-----手机号
        Password: //--用户密码
    }

getData:

    {
        Auth: //-------登录成功返回用户Auth
        UserId: //-----登录成功返回用户ID

        此处格式固定，服务器返回验证数据
        Response: {
            Time: //---服务器当前时间
            State: //--用户登录状态（200：用户登录成功；-200：用户登录失败或者未登录）
            Ack: //----返回数据状态（Success、Failure）根据这个状态来判断数据是否提交成功
        }
    }
