个人主页模块相关接口定义

1.个人主页

接口名称：getUserInfo

postData:

    {
        UserId: //------主动提交的用户ID
        Header: {
            UserId: //--当前登录用户的ID
            Cid: //-----设备Id
            Auth: //----当前登录用户的Auth
        }
    }

getData:

    {
        UserInfo:{
            UserImage: //-------用户头像
            Username: //--------用户名称
            Exper: //-----------经验值
            RankName: //--------用户等级名称
            Follow: //----------关注数
            Fans: //------------粉丝数
        },
        ArticleList:[ //--------发布（取最近更新的前6个栏目）
            {
                CateId: //------栏目ID
                ImageUrl: //----栏目封面图
                CateName: //----栏目名称
                ArticleNum: //--发布总数
                PraiseNum: //---赞的总数 
            }
        ],
        CollectList:[ //--------收藏（取最近更新的前6个栏目）
            {
                CateId: //------栏目ID
                ImageUrl: //----栏目封面图
                CateName: //----栏目名称
                ArticleNum: //--发布总数
                PraiseNum: //---赞的总数 
            }
        ],
        ClubList:[ //-----------圈子（取最近更新的前5个圈子）
            {
                ClubId: //------圈子ID
                ImageUrl: //----圈子封面图
                ClubName: //----圈子名称
                TodayNews: //---今日更新总数
            }
        ],
        CollectNum: //----------收藏总数
        ArticleNum: //----------发布总数
        ClubNum: //-------------圈子总数
    }

2.个人主页-设置|别人的主页-查看资料

接口名称：getUserSetting

postData:

    {
        UserId: //------主动提交的用户ID
        Header: {
            UserId: //--当前登录用户的ID
            Cid: //-----设备Id
            Auth: //----当前登录用户的Auth
        }
    }

getData:

    {
        UserImage: //---------用户头像
        UserName: //----------用户名称
        Sex: //---------------性别
        Job: //---------------岗位
        Area: //--------------地区

        AttentCate: [ //------关注的栏目（订阅）,只显示一级栏目
            {
                CateId: //----栏目ID
                CateName: //--栏目名称
            }
        ]
        PushState: //---------消息推送（1、容许；2、不容许）
        WhisperState: //------只允许我关注的人跟我私聊（1、容许；2、不容许）
        PhoneState: //--------允许通过手机通讯录加我好友（1、容许；2、不容许）
        PhotoState: //--------保存图片到相册（1、保存；2、不保存）
    }

3.个人主页-设置-修改密码

接口名称：setUserPassword

postData:

    {   
        UserId: //-------主动提交的用户ID
        OldPassword: //--旧密码
        NewPassword: //--新密码
        Header: {
            UserId: //---当前登录用户的ID
            Cid: //------设备Id
            Auth: //-----当前登录用户的Auth
        }
    }

4.个人主页-圈子

接口名称：getUserClub

postData:

    {
        UserId: //------主动提交的用户ID
        Header: {
            UserId: //--当前登录用户的ID
            Cid: //-----设备Id
            Auth: //----当前登录用户的Auth
        }
    }

getData:

    {
        ClubList: [
            {
                Letter: //------------圈子所属字母
                Name: //--------------圈子名称
                ImageUrl: //----------圈子封面图
                TotalUser: //---------圈子成员数目
                Category:[ //---------圈子所属栏目名称（一级、二级、三级名称）不会跨栏目
                    {
                        CateId: //----栏目ID
                        CateName: //--栏目名称
                    }
                ]
            }
        ]
    }

7.个人主页-发布

接口名称：getUserArticle

postData:

    {
        UserId: //------主动提交的用户ID 
        Header: {
            UserId: //--当前登录用户的ID
            Cid: //-----设备Id
            Auth: //----当前登录用户的Auth
        }
    }

getData:

    {
        UserImage: //-----------------用户头像 
        UserName: //------------------用户名称 
        Total: //---------------------该栏目中所有的帖子数目 
        Praise: //--------------------该栏目中所有帖子被赞过的总和 
        CateName: //------------------栏目名称
        ArticleList: [
            {
                ImageUrl: //----------图片地址
                Content: //-----------描述内容
                CollectNum: //--------被收藏的数量
                UserId: //------------用户ID
                UserImage: //---------用户头像 
                UserName: //----------用户名称
                Category:[ //---------帖子所属栏目名称（一级、二级、三级名称）不会跨栏目
                    {
                        CateId: //----栏目ID
                        CateName: //--栏目名称
                    }
                ]
            }
        ]

    }
