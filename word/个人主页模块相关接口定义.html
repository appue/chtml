个人主页模块相关接口定义

1.个人主页

接口名称：getUserInfo

postData:

    {
        
    }

getData:

    {
        UserInfo:{
            UserImage: //----用户头像
            Username: //-----用户名称
            Exper: //--------经验值
            RankName: //-----用户等级名称
            Follow: //-------关注数
            Fans: //---------粉丝数
        },
        ArticleList:[ //-----发布
            {
                ImageUrl: //--栏目封面图
                Name: //------栏目名称
                Num: //-------发布总数
                Praise: //----赞的总数 
            }
        ],
        CollectList:[ //-----收藏
            {
                ImageUrl: //--栏目封面图
                Name: //------栏目名称
                Num: //-------发布总数
                Praise: //----赞的总数 
            }
        ],
        ClubList:[ //---------圈子
            {
                ImageUrl: //--圈子封面图
                Name: //------圈子名称
                Num: //-------今日更新总数
            }
        ],
        CollectNum: //-------收藏总数
        ArticleNum: //-------发布总数
        ClubNum: //----------圈子总数
    }

2.个人主页-设置

接口名称：getUserSetting

postData:

    {
        
    }

getData:

    {
        UserImage: //-----用户头像
        UserName: //------用户名称
        Sex: //-----------性别
        Job: //-----------岗位
        Area: //----------地区
        Order: //---------订阅
        PushState: //-----消息推送
        WhisperState: //--只允许我关注的人跟我私聊
        PhoneState: //----允许通过手机通讯录加我好友
        PhotoState: //----保存图片到相册
    }

3.个人主页-设置-修改密码

接口名称：setUserPassword

postData:

    {   
        UserId: //-------用户ID
        OldPassword: //--旧密码
        NewPassword: //--新密码
    }

postData:

            }
        ]
