专题相关接口定义

专题里面的帖子用户前台是没有办法发的，只能是编辑在后台把别人发的帖子加到这个专辑里面

1.专题列表

接口名称：getListSubject

专题按时间顺序降序排列，下滑可显示更多的

postData:

    {   
        
        /*
        * 专题排序类型（默认按什么？）
        * 1、创建时间降序
        * 2、创建时间升序
        * 3、ID降序
        * 4、ID升序
        * 5、帖子数目降序
        * 6、帖子数目升序
        */
        SortType: //----专题排序类型

        PageIndex: //---当前页码
        PageSize: //----每页显示多少条记录

        Header: {
            UserId: //--当前登录用户ID（未登录传空）
            Auth: //----当前登录用户Auth（未登录传空）
        }
    }

getData:

    {   
        SubjectList: [
            {
                SubjectId: //---------专题ID
                LongName: //----------专题标题
                ShortName: //---------标题简写
                ImageUrl: //----------专题封面
                Description: //-------专题描述
                ClubId: //------------推荐的圈子ID
                ClubName: //----------推荐的圈子名称
                CreateTime: //--------专题创建的时间
                TotalArticle: //------专题内帖子总数

                /*
                * 相关主题
                * 此处由编辑后台推荐相关的标签类别，可以是一级栏目页，二级栏目页或三级栏目页
                */
                CategoryList: [
                    {
                        CateId: //----栏目ID
                        CateName: //--栏目名称
                        ImageUrl: //--栏目封面图
                    }
                ]
            }
        ]
        Total: //---------------------专题总数


        此处格式固定，服务器返回验证数据
        Response: {
            Time: //---------服务器当前时间
            State: //--------用户登录状态（200：用户登录成功；-200：用户登录失败或者未登录）
        }
    }

2.专题-内容

接口名称：getSubjectInfo

这个接口是获取某个专题的一些基础信息

postData:

    {
        SubjectId: //--专题ID

        Header: {
            UserId: //--当前登录用户ID（未登录传空）
            Auth: //----当前登录用户Auth（未登录传空）
        }
    }

getData:

    {   
        SubjectId: //---------专题ID
        LongName: //----------专题标题
        ShortName: //---------标题简写
        ImageUrl: //----------专题封面
        Description: //-------专题描述

        ClubId: //------------圈子ID（这个圈子其实是后台在添加专题是推荐的圈子，每个专题只能推荐一个）
        ClubName: //----------圈子名称

        /*
        * 相关主题
        * 此处由编辑后台推荐相关的标签类别，可以是一级栏目页，二级栏目页或三级栏目页
        */
        CategoryList: [
            {
                CateId: //----栏目ID
                CateName: //--栏目名称
                ImageUrl: //--栏目封面图
            }
        ]


        此处格式固定，服务器返回验证数据
        Response: {
            Time: //---------服务器当前时间
            State: //--------用户登录状态（200：用户登录成功；-200：用户登录失败或者未登录）
        }
    }
