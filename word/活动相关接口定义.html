活动相关接口定义

1.活动列表

接口名称：getListActivity

postData:

    {
        ActivityType: //--活动类型(0、不分类型显示所有活动列表；1、文字类型；2、图片类型)
        PageIndex: //-----当前页码
        PageSize: //------每页显示多少条记录

        UserId: //--------用户ID

        Header: {
            UserId: //--当前登录用户ID（未登录传空）
            Auth: //----当前登录用户Auth（未登录传空）
        }
    }

getData:

    {   
        ActivityList: [
            {
                ActivityId: //-----活动ID
                ActivityName: //---活动标题
                ActivityLabel: //--类别名称（可以说是一个标签吧，编辑后台随便输入的）
                ActivityType: //---活动类型(1、文字类型；2、图片类型)
                ImageUrl: //-------活动封面图
                Description: //----活动描述
                UpdateTime: //-----活动最后更新的时间（用户在活动里发表内容都会更新这个时间）
                CreateTime: //-----活动创建的时间
            }
        ]
        Total: //------------------活动总数


        此处格式固定，服务器返回验证数据
        Response: {
            Time: //--------------服务器当前时间
            State: //-------------用户登录状态（True：用户登录成功；False：用户登录失败或者未登录）
        }
    }

2.晒图活动本身的一些基本属性

接口名称：getPhotoActivityInfo

postData:

    {
        ActivityId: //--活动ID

        Header: {
            UserId: //--当前登录用户ID（未登录传空）
            Auth: //----当前登录用户Auth（未登录传空）
        }
    }

getData:

    {   
        Content: {
            ActivityId: //-----活动ID
            ActivityName: //---活动标题
            ImageUrl: //-------活动封面图
            Description: //----活动描述
            UpdateTime: //-----活动最后更新的时间（用户在活动里发表内容都会更新这个时间）
            CreateTime: //-----活动创建的时间
        }

        此处格式固定，服务器返回验证数据
        Response: {
            Time: //--------------服务器当前时间
            State: //-------------用户登录状态（True：用户登录成功；False：用户登录失败或者未登录）
        }
    }

3.文字活动本身的一些基本属性

接口名称：getAritlceActivityInfo

文字活动的本质就是帖子，只不过它是根据活动ID来调取数据库里的帖子的，一个文字活动只对应一篇帖子，帖子只有在EditState为推荐状态时才能被所有人看见，不然只能作者和关注他的人才能看见该帖子的内容

postData:

    {
        ActivityId: //---帖子ID

        Header: {
            UserId: //--当前登录用户ID（未登录传空）
            Auth: //----当前登录用户Auth（未登录传空）
        }
    }

getData:

    {   
        ArticleId: //-------------帖子ID
        Images: [ //--------------帖子里的图片
            {
                ImageUrl: //------帖子图片
                Description: //---帖子描述
            }
        ]

        Author: {
            UserId: //------------帖子发布者ID
            ImageUrl: //----------帖子发布者头像
            UserName: //----------帖子发布者名称
        }

        Affect: {
            ClubId: //------------帖子所属的圈子ID
            ClubName: //----------帖子所属的圈子名称

            SubjectId: //---------帖子所属的专题ID
            SubjectName: //-------帖子所属的专题名称

            ActivityId: //--------帖子所属的活动ID
            ActivityName: //------帖子所属的活动名称

            
            /*
            * 帖子所属栏目名称（一级、二级、三级名称）不会跨栏目，这个栏目是用户选择的
            */
            CategoryList: [
                {
                    CateId: //----栏目ID
                    CateName: //--栏目名称
                }
            ]
        }

        UpdateTime: //------------帖子最后更新的时间（用户修改帖子都会更新这个时间）
        CreateTime: //------------帖子创建的时间

        Total: {
            TotalHit: //----------帖子被浏览的总数

            StatePraise: //-------该贴是否被当前登录用户赞过（1、未；2、已）
            TotalPraise: //-------帖子被赞的总数

            StateCollect: //------该贴是否被当前登录用户收藏过（1、未；2、已）
            TotalCollect: //------帖子被收藏的总数

            TotalShare: //--------帖子被分享的总数
            TotalComment: //------帖子被评论的总数
        }

        PraiseUser: [ //----------赞过该帖子的用户
            {
                UserId: //--------用户ID
                UserName: //------用户名称
                ImageUrl: //------用户头像
            }
        ]

        EditState: //-------------帖子的发布状态（1、未推荐状态；2、已推荐）


        CommentList: [ //------------帖子最近2条评论
            {
                CommentId: //-----评论ID
                ArticleId: //-----评论关联的帖子ID
                Author: {
                    UserId: //----评论发布者ID
                    ImageUrl: //--评论发布者头像
                    UserName: //--评论发布者名称
                }

                UpdateTime: //----评论更新时间
                Content: //-------评论内容
            }
        ]

        
        /*
        * 登陆后可查看同岗位发布的浏览次数最高的帖子，最多显示6个[这里后台要根据当前登录的用户来取相关的数据]
        */
        ArticleList: [
            {
                ArticleId: //------------帖子ID
                Images: [ //-------------帖子里的图片
                    {
                        ImageUrl: //-----帖子图片
                        Description: //--帖子描述
                        Width: //--------切割出来的图片的宽度
                        Height: //-------切割出来的图片的高度
                    }
                ]

                Author: {
                    UserId: //-----------帖子发布者ID
                    ImageUrl: //---------帖子发布者头像
                    UserName: //---------帖子发布者名称
                }

                TotalHit: //-------------帖子被浏览的总数

                
                /*
                * 帖子所属栏目名称（一级、二级、三级名称）不会跨栏目，这个栏目是用户选择的
                */
                CategoryList: [
                    {
                        CateId: //-------栏目ID
                        CateName: //-----栏目名称
                    }
                ]
            }
        ]


        此处格式固定，服务器返回验证数据
        Response: {
            Time: //---------服务器当前时间
            State: //--------用户登录状态（True：用户登录成功；False：用户登录失败或者未登录）
        }
    }
