评论相关接口定义

1.评论列表

接口名称：getListComment

postData:

    {   
        
        /*
        * 排序类型（默认按什么？）
        * 1、创建时间降序
        * 2、创建时间升序
        * 3、ID降序
        * 4、ID升序
        */
        SortType: //----排序类型

        ArticleId: //---帖子ID

        PageIndex: //---当前页码
        PageSize: //----每页显示多少条记录

        Header: {
            UserId: //--当前登录用户ID（未登录传空）
            Auth: //----当前登录用户Auth（未登录传空）
        }
    }

getData:

    {   
        CommentList: [
            {
                CommentId: //-----评论ID
                ArticleId: //-----评论关联的帖子ID
                Author: {
                    UserId: //----评论发布者ID
                    ImageUrl: //--评论发布者头像
                    UserName: //--评论发布者名称
                }

                UpdateTime: //----评论更新时间
                Content: //-------评论内容
            }
        ]
        Total: //-----------------评论总数
    }
