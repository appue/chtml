注册、登录、个人主页

后台登录

接口名称：getLogin

postData:

    {   
        UserName: //---用户名
        Password: //---密码（md5）
    }

getData:

    {
        Auth: //-------返回用户的Auth
        UserId: //-----返回用户的UserId

        此处格式固定，服务器返回验证数据
        Response: {
            Time: //---服务器当前时间
            State: //--用户登录状态（True：用户登录成功；False：用户登录失败或者未登录）
            Ack: //----返回数据状态（Success、Failure）根据这个状态来判断数据是否提交成功
        }
    }

获取文章列表

接口名称：getArticleList

postData:

    {
        Review: //--推荐状态[true/false]
        CateId: //--

        ReviewHome: 是否首页推荐[true/false]

        固顶格式Header
        Header: {
            UserId: //--当前登录用户ID（未登录传空）
            Auth: //----当前登录用户Auth（未登录传空）
        }
    }
