帖子相关接口定义

帖子发完以后，只会显示在他自己的个人主页，以及关注过他的人也能在首页的关注看到，栏目里面不会立即显示出来，需要编辑推荐以后才会显示（State）

1.帖子列表

接口名称：getListArticle

postData:

    {
        CateId: //--------栏目ID
        SubjectId: //-----专题ID
        ClubId: //--------圈子ID
        ActivityId: //----活动ID

        PageIndex: //-----当前页码
        PageSize: //------每页显示多少条记录

        UserId: //--------用户ID

        Header: {
            UserId: //----当前登录用户ID
            Auth: //------当前登录用户Auth
        }
    }

getData:

    {   
        ArticleList: [
            {
                ArticleId: //------------帖子ID
                Images: [ //-------------帖子里的图片
                    {
                        ImageUrl: //-----帖子图片
                        Description: //--帖子描述
                    }
                ]

                Author: {
                    UserId: //-----------帖子发布者ID
                    ImageUrl: //---------帖子发布者头像
                    UserName: //---------帖子发布者名称
                }

                
                /*
                * 帖子所属栏目名称（一级、二级、三级名称）不会跨栏目，这个栏目是用户选择的
                */
                CategoryList: [
                    {
                        CateId: //-------栏目ID
                        CateName: //-----栏目名称
                    }
                ]

                TotalCollect: //---------帖子被收藏的总数
            }
        ]
        Total: //------------------------帖子总数
    }

2.帖子内容

接口名称：getContentArticle

帖子只有在EditState为推荐状态时才能被所有人看见，不然只能作者和关注他的人才能看见该帖子的内容

postData:

    {
        ArticleId: //---帖子ID

        Header: {
            UserId: //--当前登录用户ID
            Auth: //----当前登录用户Auth
        }
    }

getData:

    {   
        ArticleId: //-------------帖子ID
        Images: [ //--------------帖子里的图片
            {
                ImageUrl: //------帖子图片
                Description: //---帖子描述
            }
        ]

        Author: {
            UserId: //------------帖子发布者ID
            ImageUrl: //----------帖子发布者头像
            UserName: //----------帖子发布者名称
        }

        Affect: {
            ClubId: //------------帖子所属的圈子ID
            ClubName: //----------帖子所属的圈子名称

            SubjectId: //---------帖子所属的专题ID
            SubjectName: //-------帖子所属的专题名称

            ActivityId: //--------帖子所属的活动ID
            ActivityName: //------帖子所属的活动名称

            
            /*
            * 帖子所属栏目名称（一级、二级、三级名称）不会跨栏目，这个栏目是用户选择的
            */
            CategoryList: [
                {
                    CateId: //----栏目ID
                    CateName: //--栏目名称
                }
            ]
        }

        UpdateTime: //------------帖子最后更新的时间（用户修改帖子都会更新这个时间）
        CreateTime: //------------帖子创建的时间

        Total: {
            TotalHit: //----------帖子被浏览的总数
            TotalPraise: //-------帖子被赞的总数
            TotalCollect: //------帖子被收藏的总数
            TotalShare: //--------帖子被分享的总数
            TotalComment: //------帖子被评论的总数
        }

        BrowseUser: [ //----------浏览过该帖子的用户
            {
                UserId: //--------用户ID
                UserName: //------用户名称
                ImageUrl: //------用户头像
            }
        ]

        EditState: //-------------帖子的发布状态（1、未推荐状态；2、已推荐）
    }
