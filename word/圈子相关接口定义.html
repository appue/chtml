圈子相关接口定义

1.圈子列表

接口名称：getListClub

postData:

    {
        
        /*
        * 1、每个圈子当天发帖量多的排前面，每24小时会自动更新一次（取前20个）
        * 2、圈子ID降序
        * 3、圈子发帖时间降序
        * 4、圈子发帖时间升序
        * 5、字母排序（所有的圈子全部吐出）
        */
        SortType: //----圈子排序类型

        UserId: //------用户ID

        Header: {
            UserId: //--当前登录用户ID（未登录传空）
            Auth: //----当前登录用户Auth（未登录传空）
        }
    }

getData:

    {   
        ClubList: [
            {
                ClubId: //------------圈子ID
                ClubName: //----------圈子名称
                ImageUrl: //----------圈子封面图
                Description: //-------圈子简介
                TotalUser: //---------圈子成员数目
                TotalArticle: //------圈子帖子数目
                Letter: //------------圈子所属字母
                UpdateTime: //--------圈子最后更新的时间（用户在圈子里发表内容都会更新这个时间）
                CreateTime: //--------圈子创建的时间

                
                /*
                * 圈子所属栏目名称（一级、二级、三级名称）不会跨栏目，这个栏目是后台设置好的，跟圈子里的帖子没任何关系
                * 帖子发布是用户自己选择栏目
                */
                CategoryList: [
                    {
                        CateId: //----栏目ID
                        CateName: //--栏目名称
                    }
                ]
            }
        ]
        Total: //---------------------圈子总数


        此处格式固定，服务器返回验证数据
        Response: {
            Time: //------------------服务器当前时间
            State: //-----------------用户登录状态（True：用户登录成功；False：用户登录失败或者未登录）
            Ack: //-------------------返回数据状态（Success、Failure）根据这个状态来判断数据是否提交成功
        }
    }

2.圈子内容

圈子的活动是怎么来的？

接口名称：getContentClub

postData:

    {
        ClubId: //------圈子ID

        Header: {
            UserId: //--当前登录用户ID（未登录传空）
            Auth: //----当前登录用户Auth（未登录传空）
        }
    }

getData:

    {   
        ClubId: //------------圈子ID
        ClubName: //----------圈子名称
        Description: //-------圈子简介
        ImageUrl: //----------圈子图片
        
        TotalMember: //-------圈子里的成员数
        TotalSign: //---------用户签到次数
        TotalAlwaysSign: //---用户连续签到次数
        StateJoin: //---------用户是否加入圈子（True/False）
        StateSign: //---------当前用户是否已签到（True/False）


        
        ActivityList: { //-------被推荐的活动
            ActivityId: //----活动ID
            ActivityName: //--活动标题
            ActivityType: //--活动类型(1、文字类型；2、图片类型)
        }

        
        /*
        * 圈子中被收藏最多的帖子，一行最多显示10个
        */
        ArticleTop: [
            {
                ArticleId: //------------帖子ID
                Images: [ //-------------帖子里的图片
                    {
                        ImageUrl: //-----帖子图片
                        Description: //--帖子描述
                        Width: //--------切割出来的图片的宽度
                        Height: //-------切割出来的图片的高度
                    }
                ]

                Author: {
                    UserId: //-----------帖子发布者ID
                    ImageUrl: //---------帖子发布者头像
                    UserName: //---------帖子发布者名称
                }

                
                /*
                * 帖子所属栏目名称（一级、二级、三级名称）不会跨栏目，这个栏目是用户选择的
                */
                CategoryList: [
                    {
                        CateId: //-------栏目ID
                        CateName: //-----栏目名称
                    }
                ]

                TotalCollect: //---------帖子被收藏的总数
            }
        ]

        
        /*
        * 圈子精选栏目名称（一级、二级、三级名称）编辑后台设置的
        */
        CategoryList: [
            {
                CateId: //----栏目ID
                CateName: //--栏目名称
                ImageUrl: //--栏目图片
            }
        ]


        此处格式固定，服务器返回验证数据
        Response: {
            Time: //----------服务器当前时间
            State: //---------用户登录状态（True：用户登录成功；False：用户登录失败或者未登录）
            Ack: //-----------返回数据状态（Success、Failure）根据这个状态来判断数据是否提交成功
        }
    }
