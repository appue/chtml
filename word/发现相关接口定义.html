发现相关接口定义

1.发现首页接口

直接编辑后台推荐吧？

接口名称：getFindHome

postData:

    {

        Header: {
            UserId: //--当前登录用户ID（未登录传空）
            Auth: //----当前登录用户Auth（未登录传空）
        }
    }

getData:

    {   
        SubjectList: [ //--------[专题集锦]专题显示6个（不包含更多）, 可向右滑动查看
            {
                SubjectId: //----专题ID
                LongName: //-----专题标题
                ShortName: //----专题标题简写
                ImageUrl: //-----专题封面图
            }
        ]

        CategoryList: [ //-------共显示15个一级栏目
            {
                CateId: //-------栏目ID
                CateName: //-----栏目名称
            }
        ]


        此处格式固定，服务器返回验证数据
        Response: {
            Time: //-------------服务器当前时间
            State: //------------用户登录状态（True：用户登录成功；False：用户登录失败或者未登录）
        }
    }

2.发现猜你喜欢

接口名称：getFindLike

postData:

    {

        Header: {
            UserId: //--当前登录用户ID（未登录传空）
            Auth: //----当前登录用户Auth（未登录传空）
        }
    }

getData:

    {   
        /*
        * 最多显示三条
        * 每天推荐3张微信中发布的内容，由编辑自己上传，此处需要上传图片和链接
        */
        PhotoList: [
            {
                Title: //----------------图片标题
                ImageUrl: //-------------图片地址
                H5Url: //----------------H5点击跳转到的网址
                AppUrl: //---------------APP点击跳转到的网址
            }
        ]

        /*
        * 读取20条吧
        * 1、同岗位的老师发布的，并且编辑从后台推荐的
        * 2、注册时订阅的栏目中的，并且由编辑从后台推荐的
        */
        ArticleList: [
            {
                ArticleId: //------------帖子ID
                Images: [ //-------------帖子里的图片
                    {
                        ImageUrl: //-----帖子图片
                        Description: //--帖子描述
                    }
                ]

                Author: {
                    UserId: //-----------帖子发布者ID
                    ImageUrl: //---------帖子发布者头像
                    UserName: //---------帖子发布者名称
                }

                
                /*
                * 帖子所属栏目名称（一级、二级、三级名称）不会跨栏目，这个栏目是用户选择的
                */
                CategoryList: [
                    {
                        CateId: //-------栏目ID
                        CateName: //-----栏目名称
                    }
                ]

                TotalCollect: //---------帖子被收藏的总数
            }
        ]

        此处格式固定，服务器返回验证数据
        Response: {
            Time: //---------------------服务器当前时间
            State: //--------------------用户登录状态（True：用户登录成功；False：用户登录失败或者未登录）
        }
    }
