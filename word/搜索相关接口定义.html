搜索相关接口定义

1.搜索-首页

接口名称：getSearch

热门关键字是后台设置好的，前台显示

postData:

    {
    }

getData:

    {   
        KeywordList: [
            {
                Keyword: //--关键字
            }
        ]
    }

2.搜索-结果分类

接口名称：getSearchContent

搜索类型：1、帖子（搜索帖子内容关键字）2、栏目（搜索栏目名称关键字）3、圈子（搜索圈子名称关键字）4、用户（搜索用户名称关键字）

postData:

    {
        Keyword: //--用户输入的关键字
        Type: //-----搜索类型
    }

getData:

    {   
        搜索的帖子结果
        ArticleList: [
            {
                ImageUrl: //----------图片地址
                Content: //-----------描述内容
                CollectNum: //--------被收藏的数量
                UserId: //------------用户ID
                UserImage: //---------用户头像 
                UserName: //----------用户名称
                Category:[ //---------帖子所属栏目名称（一级、二级、三级名称）不会跨栏目
                    {
                        CateId: //----栏目ID
                        CateName: //--栏目名称
                    }
                ]
            }
        ]

        搜索的栏目结果
        CategoryList: [
            {
                CateId: //------------栏目ID
                CateName: //----------栏目名称
                ImageUrl: //----------栏目封面图
            }
        ]

        搜索的圈子结果
        CulbList: [
            {
                CulbId: //------------圈子ID
                CulbName: //----------圈子名称
                ImageUrl: //----------圈子封面图
                TotalUser: //---------圈子成员数目
                
                /*
                * 圈子所属栏目名称（一级、二级、三级名称）不会跨栏目，这个栏目是后台设置好的，跟圈子里的帖子没任何关系
                * 帖子发布是用户自己选择栏目
                */
                Category: [ 
                    {
                        CateId: //----栏目ID
                        CateName: //--栏目名称
                    }
                ]
            }
        ]

        搜索的用户结果
        UserList: [
            {
                UserId:
                UserName:
                ImageUrl:
                FollowState: //--
            }
        ]
    }
