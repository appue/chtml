搜索相关接口定义

1.搜索-首页

接口名称：getSearch

热门关键字是后台设置好的，前台显示

postData:

    {

        Header: {
            UserId: //--当前登录用户ID（未登录传空）
            Auth: //----当前登录用户Auth（未登录传空）
        }
    }

getData:

    {   
        
        /*
        * 热门的关键字，编辑在后台添加好的，这里直接调用显示出来
        */
        
        KeywordList: [
            {
                Keyword: //--关键字
            }
        ]


        此处格式固定，服务器返回验证数据
        Response: {
            Time: //---------服务器当前时间
            State: //--------用户登录状态（200：用户登录成功；-200：用户登录失败或者未登录）
        }
    }

2.搜索-结果分类

接口名称：getSearchContent

搜索类型：1、帖子（搜索帖子内容关键字）2、栏目（搜索栏目名称关键字）3、圈子（搜索圈子名称关键字）4、用户（搜索用户名称关键字）

postData:

    {
        Keyword: //-----用户输入的关键字
        Type: //--------搜索类型（0、全部类型；1、帖子；2、栏目；3、圈子；4、用户）

        Header: {
            UserId: //--当前登录用户ID（未登录传空）
            Auth: //----当前登录用户Auth（未登录传空）
        }
    }

getData:

    {   
        搜索的帖子结果
        ArticleList: [
            {
                ArticleId: //------------帖子ID

                Images: [ //-------------帖子里的图片
                    {
                        ImageUrl: //-----帖子图片
                        Description: //--帖子描述
                    }
                ]

                TotalCollect: //---------被收藏的数量

                Author: {
                    UserId: //-----------帖子发布者ID
                    ImageUrl: //---------帖子发布者头像
                    UserName: //---------帖子发布者名称
                }

                CategoryList:[ //--------帖子所属栏目名称（一级、二级、三级名称）不会跨栏目
                    {
                        CateId: //-------栏目ID
                        CateName: //-----栏目名称
                    }
                ]
            }
        ]

        搜索的栏目结果
        CategoryList: [
            {
                CateId: //------------栏目ID
                CateName: //----------栏目名称
                ImageUrl: //----------栏目封面图
            }
        ]

        搜索的圈子结果
        ClubList: [
            {
                ClubId: //------------圈子ID
                ClubName: //----------圈子名称
                ImageUrl: //----------圈子封面图
                TotalUser: //---------圈子成员数目
                
                
                /*
                * 圈子所属栏目名称（一级、二级、三级名称）不会跨栏目，这个栏目是后台设置好的，跟圈子里的帖子没任何关系
                * 帖子发布是用户自己选择栏目
                */
                Category: [ 
                    {
                        CateId: //----栏目ID
                        CateName: //--栏目名称
                    }
                ]
            }
        ]

        搜索的用户结果
        UserList: [
            {
                UserId: //------------用户ID
                UserName: //----------用户名称
                ImageUrl: //----------用户头像
                FollowState: //-------用户关注状态（1、已关注；2、互关注；3、未关注）
            }
        ]


        此处格式固定，服务器返回验证数据
        Response: {
            Time: //------------------服务器当前时间
            State: //-----------------用户登录状态（200：用户登录成功；-200：用户登录失败或者未登录）
        }
    }
